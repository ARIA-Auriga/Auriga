		close;
	}
	mes "[スキル屋]";
	mes "覚えるスキルを選んでください";
	next;
	if(@menu == 1) {
		switch(select("チャージアタック","シュリンク")) {
		case 1:
			if(Class == Job_Knight)
				skill 1001,1,0;
			close;
		case 2:
			if(Class == Job_Crusader)
				skill 1002,1,0;
			close;
		}

		switch(select("サイトブラスター","コンバーター製造","水属性エレメンタルチェンジ","地属性エレメンタルチェンジ","火属性エレメンタルチェンジ","風属性エレメンタルチェンジ")) {

			if(Class == Job_Wizard)

			close;

			if(Class == Job_Sage)
				skill 1007,1,0;
			close;
		case 3:
			set '@skillid,1008;
			break;
		case 4:
			set '@skillid,1017;
			break;
		case 5:
			set '@skillid,1018;
			break;
		case 6:
			set '@skillid,1019;
			break;
		}
		// 3〜6のとき続き
		if(Class == Job_Sage && getskilllv(1008) <= 0 && getskilllv(1017) <= 0 && getskilllv(1018) <= 0 && getskilllv(1019) <= 0)
			skill '@skillid,1,0;
		close;
	}
	else if(@menu == 3) {
		switch(select("ファンタスミックアロー","パンボイス","魅惑のウィンク")) {
		case 1:
			if(Class == Job_Hunter)
				skill 1009,1,0;
			close;
		case 2:
			if(Class == Job_Bard)
				skill 1010,1,0;
			close;
		case 3:
			if(Class == Job_Dancer)
				skill 1011,1,0;
			close;
		}
	}
	else if(@menu == 4) {
		switch(select("レデムプティオ","気注入","気爆発")) {
		case 1:
			if(Class == Job_Priest)
				skill 1014,1,0;
			close;
		case 2:
			if(Class == Job_Monk)
				skill 1015,1,0;
			close;
		case 3:
			if(Class == Job_Monk)
				skill 1016,1,0;
			close;
		}
	}

			close;
		case 2:
			if(Class == Job_Assassin)
				skill 1004,1,0;
			close;
		case 3:
			if(Class == Job_Rogue)
				skill 1005,1,0;
			close;
		}
	}
	else if(@menu == 6) {
		switch(select("アンフェアリートリック","貪欲")) {
		case 1:
			if(Class == Job_Blacksmith)
				skill 1012,1,0;
			close;
		case 2:
			if(Class == Job_Blacksmith)
				skill 1013,1,0;
			close;
		}
	}
	close;
